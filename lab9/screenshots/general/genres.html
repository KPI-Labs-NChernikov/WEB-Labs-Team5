public async Task<IEnumerable<GenreModel>> GetMovieGenresAsync(int movieId)
        {
            return await Task.Run(() => _mapper.Map<IEnumerable<GenreModel>>(
                _context.Genres
                .Include(g => g.Movies)
                .Include(g => g.Users)
                .Where(r => r.Movies.Select(m => m.Id).Contains(movieId))));
        }
