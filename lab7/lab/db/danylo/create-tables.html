CREATE TABLE Train 
( 
    id SMALLINT UNSIGNED AUTO_INCREMENT PRIMARY KEY NOT NULL ,
    DispatchStation VARCHAR(50) NOT NULL,
    ArrivalStation VARCHAR(50) NOT NULL,
    NumberOfCars INT NOT NULL,
    DepartureTime VARCHAR(5) NOT NULL,
    ArrivalTime VARCHAR(5) NOT NULL
);

CREATE TABLE Schedule 
( 
    id SMALLINT UNSIGNED AUTO_INCREMENT PRIMARY KEY NOT NULL,
    TrainNumber SMALLINT UNSIGNED NOT NULL,
    DateOfDeparture VARCHAR(10) NOT NULL,
    NumberOfTickets INT NOT NULL,
    TicketPrice INT NOT NULL,
    FOREIGN KEY fkTrain(TrainNumber)
    REFERENCES Train(Id)
    ON UPDATE CASCADE
    ON DELETE RESTRICT
);
